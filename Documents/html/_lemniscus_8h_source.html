341  Vector3f GetPosition1(float theta) {
342 
343  const float p = initRadius;
344 
345  float cos2theta;
346 
347  Vector3f pos;
348 
349  if ((cos2theta = cos(2*theta)) < 0) {
350  pos.x = p*sqrt(abs(cos2theta))*cos(theta);
351  pos.y = p*sqrt(abs(cos2theta))*sin(theta);
352  } else {
353  pos.x = -p*sqrt(cos2theta)*cos(theta);
354  pos.y = -p*sqrt(cos2theta)*sin(theta);
355  }
356  pos.z = GetZ(pos.x, pos.y);
357  return pos;

366  Vector3f GetPosition2(float theta) {
367 
368  const float p = initRadius;
369 
370  static float oldtheta = 0;
371  static int state = 0;
372  float cosin, sinin;
373  cosin = cos(theta);
374  sinin = sin(theta);
375 
376  Vector3f pos;
377 
378 
379  if (theta < D3DX_PI/2 || theta > 3*D3DX_PI/2) {
380  pos.x = 2*p*cosin*cosin;
381  pos.y = 2*p*cosin*sinin;
382  } else {
383  pos.x = -2*p*cosin*cosin;
384  pos.y = 2*p*cosin*sinin;
385  }
386 
387  if ((theta < Elapse || theta > D3DX_PI*2-Elapse) || (oldtheta < D3DX_PI+Elapse && theta > D3DX_PI-Elapse )) {
388  state = (++state)%4;
389  printf("oldtheta(%0.2f), theta(%0.2f), state(%d)\n", oldtheta, theta, state);
390  }
391 
392  pos.z = GetZ(pos.x, pos.y, state);
